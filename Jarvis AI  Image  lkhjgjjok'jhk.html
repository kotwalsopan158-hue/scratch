def clean_text_for_speech(text):
    return re.sub(r'[^\w\s,.?!]', '', text)

    if muted or not activated:

    log(text, overwrite=True)

    engine.say(clean)

def handle_weather_query(query):
    data = get_weather_data()
    if not data:
        speak("Sorry, I couldn't fetch the weather data.")
        return

    query = query.lower()

    if "today" in query:
        temp = data["today"]
        cond = get_condition(temp["weather_code"])
        speak(
            f"Today's weather in Georai is {cond}, with a high of {temp['temp_max']}В°C and wind speed of {temp['wind']} km/h.")
    elif "tomorrow" in query:
        temp = data["tomorrow"]
        cond = get_condition(temp["weather_code"])
        speak(
            f"Tomorrow's weather in Georai will be {cond}, with a high of {temp['temp_max']}В°C and wind speed of {temp['wind']} km/h.")
    elif "yesterday" in query:
        temp = data["yesterday"]
        cond = get_condition(temp["weather_code"])
        speak(
            f"Yesterday in Georai was {cond}, with a high of {temp['temp_max']}В°C and wind speed of {temp['wind']} km/h.")
    elif "wind speed" in query or "wind" in query:
        wind = data["today"]["wind"]
        speak(f"Today's wind speed in Georai is {wind} kilometers per hour.")
    elif "temperature" in query:
        temp = data["today"]
        speak(f"Today's temperature in Georai is around {temp['temp_max']} degrees Celsius.")
    elif "weather" in query:
        temp = data["today"]
        cond = get_condition(temp["weather_code"])
        speak(f"Today's weather in Georai is {cond}.")
    else:
        speak("Please ask about today's, yesterday's or tomorrow's weather or wind.")

# Close app function
def cut_app(app_name):
    key = app_name.lower().strip()
    process_names = {
        "chatgpt": "ChatGPT.exe",
        "ai": "ChatGPT.exe",
        "whatsapp": "WhatsApp.exe",
        "this pc": "explorer.exe",
        "microsoft": "msedge.exe",
        "video editor": "openshot-qt.exe",
        "mobile cast": "AnyDesk.exe",
        "chrome": "chrome.exe"
    }
    if key in process_names:
        pname = process_names[key]
        killed = False
        for proc in psutil.process_iter(['pid', 'name']):
            if proc.info['name'] and proc.info['name'].lower() == pname.lower():
                try:
                    proc.kill()
                    killed = True
                except:
                    pass
        speak(f"{key.title()} closed." if killed else f"{key.title()} was not running.")
    else:
        speak(f"No process info for {app_name}")

# ============ Q&A & CALCULATOR ============
def answer_from_wikipedia(query):
    try:
        return wikipedia.summary(query, sentences=2)
    except:
        return None


def answer_from_duckduckgo(query):
    try:
        with DDGS() as ddgs:
            results = ddgs.text(query)
            for result in results:
                return result.get('body') or result.get('snippet')
    except:
        return None


def answer_question(query):
    result = answer_from_wikipedia(query) or answer_from_duckduckgo(query)
    if result:
        speak(result)
    else:
        speak("Sorry, I couldn't find an answer.")


def format_number_indian(num):
    num_str = str(num)
    if '.' in num_str:
        integer_part, decimal_part = num_str.split('.')
    else:
        integer_part, decimal_part = num_str, None
    rev = integer_part[::-1]
    parts = [rev[:3]] + [rev[i:i + 2] for i in range(3, len(rev), 2)]
    formatted = ','.join(parts)[::-1]
    return formatted + ('.' + decimal_part if decimal_part else '')


def evaluate_calculation(query):
    try:
        result = eval(query.replace('x', '*').replace('X', '*'))
        speak("The result is " + format_number_indian(result))
    except:
        speak("Sorry, I couldn't calculate that.")

# ============ HANDLE COMMAND ============
def handle_command(command):
    global muted, activated
    command = command.lower()


    if "activate" in command and "mute" not in command:
        activated = True
        muted = False
        set_status("рџџў Activated", "lightgreen")
        speak("Activated.")
        return

    if "deactivate" in command and "mute" not in command:
        activated = False
        muted = True
        set_status("вќЊ Deactivated", "red")
        log("вќЊ Deactivated.", overwrite=True)
        return

    if not activated:
        return

    if "mute activate" in command or command.strip() == "mute":
        muted = True
        set_status("рџ”‡ Muted", "orange")
        log("рџ”‡ Mute activated.", overwrite=True)
        return
    elif "mute deactivate" in command or "unmute" in command:
        muted = False
        set_status("рџ”Љ Unmuted", "lightgreen")
        log("рџ”Љ Mute deactivated.", overwrite=True)
        speak("Unmuted.")
        return

    if muted:
        return

    if command in ["goodbye", "exit", "bye-bye"]:
        speak("Goodbye!")
        root.quit()
        sys.exit()

    elif "what is my name" in command:
        speak("Your name is Samarth")
    elif "where do i live" in command:
        speak("You live in Georai")
    elif "what is my birth date" in command:
        speak("Your birth date is 24 April 2013")
    elif "what is my age" in command or "how old am i" in command:
        speak("You are 13 years old")
    elif "joke" in command:
        speak(random.choice(jokes))
    elif command.startswith("open "):
        app_name = command.replace("open ", "").strip()
        open_app(app_name)
    elif command.startswith("close ") or command.startswith("cut "):
        app_name = command.replace("close ", "").replace("cut ", "").strip()
        cut_app(app_name)





    elif "time" in command:
        speak(datetime.datetime.now().strftime("%I:%M %p"))
    elif "date" in command:
        speak(datetime.datetime.now().strftime("%A, %d %B %Y"))
    elif "search youtube for" in command:
        youtube_search(command)
    elif "search google for" in command or "search for" in command:
        google_search(command)
    elif command.startswith("calculate"):
        expression = command.replace("calculate", "").strip()
        evaluate_calculation(expression)
    elif any(command.startswith(w) for w in ["what", "where", "when", "who", "why", "how"]):
        if any(word in command for word in ["weather", "temperature", "wind", "rain", "snow"]):
            handle_weather_query(command)
        else:
            answer_question(command)
    elif any(word in command for word in ["shutdown", "restart", "sleep mode", "lock", "login"]):
        system_control(command)










# ============ LISTEN LOOP ============
def listen_loop():
    recognizer = sr.Recognizer()
    mic = sr.Microphone()
    with mic as source:
        recognizer.adjust_for_ambient_noise(source, duration=2)
        set_status("рџЋ¤ Listening...", "skyblue")  # Keep listening animation here
    while True:
        try:
            with mic as source:
                set_status("рџЋ¤ Listening...", "skyblue")  # Keep listening animation
                audio = recognizer.listen(source, timeout=5, phrase_time_limit=8)
                set_status("рџ§  Recognizing...", "cyan")  # <-- REMOVE or COMMENT THIS OUT
                command = recognizer.recognize_google(audio)
                log(f"рџ§‘ You: {command}", overwrite=True)
                handle_command(command)
        except sr.WaitTimeoutError:
            set_status("рџЋ¤ Listening...", "skyblue")  # Keep listening animation
            continue
        except sr.UnknownValueError:
            set_status("рџЋ¤ Listening...", "skyblue")  # Keep listening animation
            continue
        except sr.RequestError:
            speak("Speech service unavailable")



# ============ SHUTDOWN, RESTART, SLEEP, LOCK ============
def system_control(command):
    command = command.lower()
    if "shutdown" in command:
        speak("Shutting down the system.")
        os.system("shutdown /s /t 1")
    elif "restart" in command:
        speak("Restarting the system.")
        os.system("shutdown /r /t 1")
    elif "sleep" in command:
        speak("Putting the system to sleep.")
        os.system("rundll32.exe powrprof.dll,SetSuspendState 0,1,0")
    elif "lock" in command or "login" in command:
        speak("Locking the system.")
        ctypes.windll.user32.LockWorkStation()
